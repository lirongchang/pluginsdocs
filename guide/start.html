# 快速上手

# 前提条件

需按构建顺序进行插件的调用，插件调用顺序如下：

# 构建前调用

codecheck的动态配置必须在构建前完成。
同理其他插件的动态配置同样可以在构建前进行调用。

# 构建时调用

mavenbuild插件即为构建时调用。
待构建的maven工程的pom.xml文件中必须包含junit单元测试依赖，checkstyle静态代码检查依赖，jacoco覆盖率检查依赖。
否则无法构建。 调用插件的工程也需要进行相关的配置。后面会详细介绍相关的pom.xml的配置。

# 构建后调用

构建后主要进行分析，但是需要先进行动态配置，然后进行分析。
passingcheck分析和coveragecheck分析以及codecheck分析不分先后。
scoringanalysis分数分析基于codecheck的分析数据和passingcheck的分析数据以及coveragecheck的分析数据，所以须在codecheck、passingcheck和coveragecheck分析之后进行分析。
codeduplicate检查须在所有学生作业提交之后进行统一查重。
summaryanalysis则在最后进行分析。

# 待构建工程

待构建工程需要有下述的maven依赖，同时还应按照插件开发时制定的目录规范来设计学生工程的存放路径。

# mavenbuild

在使用mavenbuild插件进行构建前，应确保待构建工程包含如下的maven依赖： 建议直接复制以下pom.xml内容作为待构建工程的标准pom.xml文件，其中的依赖的版本号不建议使用比下面的版本更低的版本号。

<properties>
		<project.build.sourceEncoding>UTF-8</project.build.sourceEncoding>
		<project.reporting.outputEncoding>UTF-8</project.reporting.outputEncoding>
		<!-- java版本号 -->
		<java.version>1.8</java.version>
		<!-- junit版本号 -->
		<junit.version>4.12</junit.version>
		<!-- jacoco版本号 -->
		<jacoco.version>0.8.3</jacoco.version>
        <!-- checkstyle版本号 -->
        <maven.checkstyle.version>3.0.0</maven.checkstyle.version>
        <maven.jxr.version>3.0.0</maven.jxr.version>
        <!-- maven版本 -->
        <maven.version>3.6.1</maven.version>
		<!-- maven-surefire插件版本号 -->
		<maven.surefire.plugin.version>2.5</maven.surefire.plugin.version>
	</properties>

	<dependencies>
		<!--单元测试依赖 -->
		<dependency>
			<groupId>junit</groupId>
			<artifactId>junit</artifactId>
			<version>${junit.version}</version>
			<scope>test</scope>
		</dependency>
		<!-- jacoco依赖 -->
		<dependency>
            <groupId>org.jacoco</groupId>
            <artifactId>jacoco-maven-plugin</artifactId>
            <version>${jacoco.version}</version>
		</dependency>
	</dependencies>

	<build>
		<plugins>
			<!-- maven插件配置 -->
			<plugin>
				<groupId>org.apache.maven.plugins</groupId>
				<artifactId>maven-compiler-plugin</artifactId>
				<version>${maven.version}</version>
				<configuration>
					<skipMain>true</skipMain>
					<skip>true</skip>
					<source>${java.version}</source>
					<target>${java.version}</target>
					<encoding>${project.build.sourceEncoding}</encoding>
				</configuration>
			</plugin>
			
			<!-- maven surefire插件配置 -->
			<plugin>
	            <groupId>org.apache.maven.plugins</groupId>
	            <artifactId>maven-surefire-plugin</artifactId>
	            <version>${maven.surefire.plugin.version}</version>
	            <configuration>
	                <skipTests>false</skipTests>
	                <argLine>${argLine} -Dfile.encoding=UTF-8</argLine>
                     <!-- 忽略测试失败 -->
	                <testFailureIgnore>true</testFailureIgnore>
	            </configuration>
        	</plugin>
        
			<!-- 测试覆盖率插件配置 -->
			<plugin>
				<groupId>org.jacoco</groupId>
				<artifactId>jacoco-maven-plugin</artifactId>
				<version>${jacoco.version}</version>
				<executions>
					<execution>
						<id>prepare-agent</id>
						<goals>
							<goal>prepare-agent</goal>
						</goals>
					</execution>
					<execution>
						<id>report</id>
						<phase>prepare-package</phase>
						<goals>
							<goal>report</goal>
						</goals>
					</execution>
					<execution>
						<id>post-unit-test</id>
						<phase>test</phase>
						<goals>
							<goal>report</goal>
						</goals>
						<configuration>
							<dataFile>target/jacoco.exec</dataFile>
							<outputDirectory>target/coverage-reports</outputDirectory>
						</configuration>
					</execution>
				</executions>
				<configuration>
					<skipTests>false</skipTests>
					<destFile>target/jacoco.exec</destFile>
				</configuration>
			</plugin>
			<plugin>
				<groupId>org.apache.maven.plugins</groupId>
				<artifactId>maven-checkstyle-plugin</artifactId> 
				<version>${maven.checkstyle.version}</version>
					<configuration>
						<configLocation>checkstyle.xml</configLocation>
		                 <sourceDirectories></sourceDirectories>
		                 <testSourceDirectories>${project.build.testSourceDirectory}</testSourceDirectories>
		                 <includeTestSourceDirectory>true</includeTestSourceDirectory>
					</configuration> 
			</plugin>
		</plugins>
	</build>
	
	<reporting>  
  		<plugins> 
         <plugin>
            <groupId>org.apache.maven.plugins</groupId>
            <artifactId>maven-checkstyle-plugin</artifactId>
            <version>${maven.checkstyle.version}</version>
            <inherited>false</inherited>
            <reportSets>
              <reportSet>
                <reports>
                  <report>checkstyle-aggregate</report>
                </reports>
              </reportSet>
            </reportSets>
          </plugin>
           <plugin>  
             <groupId>org.apache.maven.plugins</groupId>  
              <artifactId>maven-jxr-plugin</artifactId>  
               <version>${maven.jxr.version}</version>  
          </plugin>  
        </plugins>  
    </reporting>  
</project>

# 学生作业目录设计

windows平台：
D:/students/年份-1/班级名称/学生学号/待构建工程

└─2019-1

        ├─5120150041

# 插件调用工程

插件调用工程同样需要配置maven依赖，才能调用插件而不至于出错。 下面是插件调用工程需要配置的maven项目的pom.xml文件中的相关依赖项：

<properties>
		<project.build.sourceEncoding>UTF-8</project.build.sourceEncoding>
		<project.reporting.outputEncoding>UTF-8</project.reporting.outputEncoding>
		<!-- java版本号 -->
		<java.version>1.8</java.version>
		<!-- maven版本 -->
        <maven.version>3.8.0</maven.version>
		<!-- xml文件处理 -->
		<dom4j.version>1.6.1</dom4j.version>
		<!-- jsoup -->
		<jsoup.version>1.11.2</jsoup.version>
		<!-- json版本 -->
		<json.version>20180130</json.version>
		<jackjson.version>2.9.8</jackjson.version>
		<!-- java处理xml文件的依赖 -->
		<jdom.version>1.0</jdom.version>
		<commons.beanutils.version>1.9.3</commons.beanutils.version>
		<!-- jfreechart -->
		<jfreechart.version>1.5.0</jfreechart.version>
	</properties>

	<dependencies>
		<dependency>
		    <groupId>dom4j</groupId>
		    <artifactId>dom4j</artifactId>
		    <version>${dom4j.version}</version>
		</dependency>
		<!-- html文件解析相关 -->
		<dependency>
		    <groupId>org.jsoup</groupId>
		    <artifactId>jsoup</artifactId>
		    <version>${jsoup.version}</version>
		</dependency>
		<!-- json数据处理相关 -->
		<dependency>
		    <groupId>org.json</groupId>
		    <artifactId>json</artifactId>
		    <version>${json.version}</version>
 		</dependency>
		<!-- jackjson -->
		<dependency>
		    <groupId>com.fasterxml.jackson.core</groupId>
		    <artifactId>jackson-databind</artifactId>
		    <version>${jackjson.version}</version>
		</dependency>
		<dependency>
		  <groupId>com.fasterxml.jackson.core</groupId>
		   <artifactId>jackson-core</artifactId>
		   <version>${jackjson.version}</version>
		</dependency>
		<dependency>
		   <groupId>com.fasterxml.jackson.core</groupId>
		   <artifactId>jackson-annotations</artifactId>
		   <version>${jackjson.version}</version>
		</dependency>
		<dependency>
		      <groupId>jdom</groupId>
		      <artifactId>jdom</artifactId>
		      <version>${jdom.version}</version>
		</dependency>
		<dependency>
		      <groupId>commons-beanutils</groupId>
		      <artifactId>commons-beanutils</artifactId>
		      <version>${commons.beanutils.version}</version>
		</dependency>
		<dependency>
		    <groupId>org.jfree</groupId>
		    <artifactId>jfreechart</artifactId>
		    <version>${jfreechart.version}</version>
		</dependency>
	</dependencies>

	<build>
		<plugins>
			<!-- maven插件配置 -->
			<plugin>
				<groupId>org.apache.maven.plugins</groupId>
				<artifactId>maven-compiler-plugin</artifactId>
				<version>${maven.version}</version>
				<configuration>
					<skipMain>true</skipMain>
					<skip>true</skip>
					<source>${java.version}</source>
					<target>${java.version}</target>
					<encoding>${project.build.sourceEncoding}</encoding>
				</configuration>
			</plugin>
		</plugins>
	</build>

</project>

# 具体调用

首先下载插件，然后进行调用

//静态调用：
import 包名.类名
//然后直接新建对象调用类对象方法
//例如：
import com.mavenbuild.main.impl.MavenBuildInterfaceImpl;
MavenBuildInterfaceImpl build=new MavenBuildInterfaceImpl();
String result=build.executeMavenBuild(projectPath);

//动态加载调用：
//首先加载外部jar包，然后返回待调用jar包中的方法，最后执行jar包指定类中的指定的方法
//例如：
    private static final String jarPackageNamePath="D:/plugins/mavenbuild.jar";  //jar包存放在D:/plugins目录下
	private static final String implClassPath="com.mavenbuild.main.impl";        //待执行的包名
	private static final String classPath=implClassPath+".MavenBuildInterfaceImpl";  //待执行的包中的类名
	private static Class<?> build;

	/**
	 * 进行maven构建，带一个参数，有返回值
	 */
	public String executeMavenBuild(String projectPath) {
		build=PluginService.getClass(jarPackageNamePath, classPath);
		String methodName="executeMavenBuild";   //待执行的方法名
		Method method=PluginService.getMethod(build, methodName);
		String result=PluginService.executeMethod(build, method,projectPath);
		return result;
	}

    PluginService类中的方法如下：
    /**
	 * 加载外部jar包
	 * @param jarPath jar包的存放路径
	 * @param classPath 待加载的类在jar包中的路径
	 * @return 返回待调用的类
	 */
	public static Class<?> getClass(String jarPath,String classPath){
		try {
			String urlPath="file:"+jarPath;
			URL url=new URL(urlPath);
			URLClassLoader urlClassLoader=new URLClassLoader(new URL[] {url},Thread.currentThread().getContextClassLoader());
			Class<?> clazz=urlClassLoader.loadClass(classPath);
			return clazz;
		} catch (Exception e) {
			e.printStackTrace();
		}
		return null;
	}
    /**
	 * 返回待调用jar包中的方法,带1个字符串参数
	 * @param clazz 待调用的类
	 * @param methodName 待调用的方法
	 * @return 返回待执行的方法
	 */
	public static Method getMethod(Class<?> clazz,String methodName) {
		try {
			Method method=clazz.getMethod(methodName,String.class);  //得到方法对象,有参的方法需要指定参数类型
			return method;
		} catch (Exception e) {
			e.printStackTrace();
		}
		return null;
	}
    /**
	 * 执行jar包指定类中的指定方法，带1个参数，返回值为字符串形式
	 * @param clazz 待调用的类
	 * @param method 待调用的方法
	 * @param argument 待调用的参数
	 * @return 调用结果
	 */
	public static String executeMethod(Class<?> clazz,Method method,String argument) {
		String result=null;
		try {
			result=(String) method.invoke(clazz.newInstance(),argument);
		}catch (Exception e) {
			e.printStackTrace();
		}
		return result;
	}
