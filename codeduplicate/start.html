# 快速上手

# 插件接口

包名：com.codeduplicate.main.api; 接口名：DuplicateConfig;

接口方法 方法参数 返回值 方法说明
String provideHtmlSetting(); 返回值为html页面中的div标签内的内容 提供在html页面可以进行配置的相关参数
String getHtmlSetting(String content); content为配置后的json字符串 返回值为设置过的html页面中的div标签内的包含参数值的字符串 获取在html页面所配置的检查项信息，进行配置后返回html中的div
String getScoreSetting(String content); content为配置后的json字符串 返回值为分数设置文件的字符串内容 获取配置后的分数设置

包名：com.codeduplicate.main.api; 接口名：DuplicateAnalyse;

接口方法 方法参数 返回值 方法说明
String executeDuplicateCheck(String semester,String projectName); semester为学期路径，projectName为待进行查重的工程名 失败则返回"error"，否则返回查重的结果路径 执行代码查重
String getSimilarity(String semester,String projectName); semester为学期路径，projectName为待进行查重的工程名 返回值为每个学生的学号和最大相似度 获取最大相似度
String getDeatilSimilarity(String semester,String projectName); semester为学期路径，projectName为待进行查重的工程名 返回值为每个学生提交的作业与其他作业进行查重比较后的相似度字符串 获取每个学生的详细的相似度信息

# 接口实现类

上述接口的实现类分别为：
包名：com.codeduplicate.main.impl; 实现类名：DuplicateConfigImpl;
对应的实现类中的方法和接口中定义的一样，在此不在赘述。

包名：com.codeduplicate.main.impl; 实现类名：DuplicateAnalyseImpl;
对应的实现类中的方法和接口中定义的一样，在此不在赘述。

# 调用举例

静态调用举例如下，动态加载调用回看插件使用指南中的举例。
代码查重插件动态配置接口的调用举例：

public class DuplicateConfigImplTest {

	DuplicateConfigImpl config;
    //页面上动态配置后的json字符串
	String json1;
	String json2;
	
	
	@BeforeClass
	public static void setUpBeforeClass() throws Exception {
	}

	@AfterClass
	public static void tearDownAfterClass() throws Exception {
	}

	@Before
	public void setUp() throws Exception {
		config=new DuplicateConfigImpl();
		json1="{\"similarityrate\":\"100\"}\r\n";
		json2="{\"similarityrate\":\"80\"}\r\n";
	}

	@After
	public void tearDown() throws Exception {
	}

	@Test
	public void testGetHtmlSetting() {
        //获取配置后包含配置信息的html页面中的div标签内容
		String result=config.getHtmlSetting(json2);
		Assert.assertNotNull(result);
		System.out.println(result);
	}

	@Test
	public void testProvideHtmlSetting() {
        //获取页面动态配置检查项html页面中的div标签内容
		String result=config.provideHtmlSetting();
		Assert.assertNotNull(result);
		System.out.println(result);
	}

	@Test
	public void testGetScoreSetting() {
        //获取动态配置后的相似度设置信息
		String result=config.getScoreSetting(json1);
		Assert.assertNotNull(result);
		System.out.println(result);
	}
}

代码查重分析接口调用举例：

public class DuplicateAnalyseImplTest {

	DuplicateAnalyseImpl analyseService;
	//待进行代码查重的工程名
	String projectName1;
	String projectName2;
	String semester;  //学期路径
	@BeforeClass
	public static void setUpBeforeClass() throws Exception {
	}

	@AfterClass
	public static void tearDownAfterClass() throws Exception {
	}

	@Before
	public void setUp() throws Exception {
		analyseService=new DuplicateAnalyseImpl();
		projectName1="nextday";
		projectName2="job1";
		semester="D:/students/2019-1/";
	}

	@After
	public void tearDown() throws Exception {
	}

	@Test
	public void testExecuteDuplicateCheck() {
		//调用方法进行对"nextday"作业工程进行代码查重，查重成功返回查重结果的保存路径，否则返回error
		String result=analyseService.executeDuplicateCheck(semester,projectName1);
		System.out.println(result);
	}

	@Test
	public void testGetSimilarity() {
		//调用此方法进行对"nextday"作业工程进行代码查重分析，获取每个学生的最大相似度，以字符串形式返回
		String result=analyseService.getSimilarity(semester,projectName1);
		System.out.println(result);
	}

	@Test
	public void testGetDeatilSimilarity() {
		//调用此方法对"nextday"作业工程进代码查重分析，获取每个学生的代码查重的详细信息
		String result=analyseService.getDeatilSimilarity(semester,projectName1);
		System.out.println(result);
	}
}
