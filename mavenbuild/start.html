# 快速上手

# 插件接口

包名：com.mavenbuild.main.api; 接口名：MavenBuildInterface; 以下方法是接口中的核心方法，其他方法就不再列举了。

接口方法 方法参数 返回值 方法说明
String executeMavenBuild(String projectPath); 工程路径名 already built,no build 执行maven构建
String executeMavenBuild(String projectPath,String mavenCommand) 工程路径名,maven命令 already built,no build 执行maven构建
String getMavenExecuteResult(String projectPath); 工程路径名 success,error,no build 获取maven构建状态

# 接口实现类

包名：com.mavenbuild.main.impl; 实现类名：MavenBuildInterfaceImpl; 对应的实现类中的方法和接口中定义的一样，在此不在赘述。

# 调用举例

静态调用举例如下，动态加载调用回看插件使用指南中的举例。

public class MavenBuildInterfaceImplTest {
    MavenBuildInterfaceImpl test;
	String projectPath;
	
	@Before
	public void setUp() throws Exception {
		test=new MavenBuildInterfaceImpl();
		projectPath="D:/testplugin/sort";  //待执行的工程路径名
	}

	@Test
	public void testExecuteMavenBuild() throws IOException {
        //使用默认的maven命令，执行maven构建
	    String result=test.executeMavenBuild(projectPath);
        System.out.println(result);
	}
	
	@Test
	public void testExecuteMavenBuild2() {
		String mavenCommand="mvn clean test -Dmaven.clean.failOnError=false checkstyle:checkstyle  jxr:jxr install ";
		try {
            //执行自定义的maven命令
			String result=test.executeMavenBuild(projectPath, mavenCommand);
            System.out.println(result);
		} catch (IOException e) {
			e.printStackTrace();
		}
	}
	
	@Test
	public void testGetMavenExecuteResult() {
		try {
            //获取maven构建后的构建状态
			String result=test.getMavenExecuteResult(projectPath);
			System.out.println(result);
		} catch (IOException e) {
			e.printStackTrace();
		}
	}
